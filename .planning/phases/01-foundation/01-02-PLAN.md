```python
from datetime import datetime, timezone

# ENG-01: Initial confidence — Beta(2,1) prior. First failure (0.67 - 0.10 = 0.57) stays
# in Experimental tier; 4 successes reach Gold (0.67→0.72→0.77→0.82→0.87).
INIT_CONFIDENCE = 0.67

# ENG-02: Delta constants
SUCCESS_DELTA = 0.05
FAILURE_DELTA = -0.10
MAX_CONFIDENCE = 0.95
MIN_CONFIDENCE = 0.0

# ENG-05: Consecutive failures required to demote a Gold-tier pattern
GOLD_EXIT_THRESHOLD = 3


def classify_tier(confidence: float) -> str:
    """ENG-03: Classify confidence value into tier string.

    Thresholds (inclusive lower bound):
      Gold        >= 0.85
      Production  >= 0.70
      Experimental >= 0.50
      Demoted     < 0.50
    """
    if confidence >= 0.85:
        return "gold"
    if confidence >= 0.70:
        return "production"
    if confidence >= 0.50:
        return "experimental"
    return "demoted"


def apply_outcome(pattern: dict, outcome: str) -> dict:
    """Return updated pattern dict after applying outcome.

    Pure function — no I/O, no side effects. Returns a NEW dict (does not mutate input).

    Evaluation order is CRITICAL — must not be changed:
      Step 1: demotion lock (ENG-04) — demoted patterns never auto-promote
      Step 2: Gold exit guardrail (ENG-05) — requires 3 consecutive failures to leave Gold
      Step 3: classify_tier from new confidence (ENG-03)

    Args:
        pattern: dict with keys: id, confidence, tier, consecutive_failures, times_applied
        outcome: "success" or "failure"

    Returns:
        New dict with updated confidence, tier, consecutive_failures, times_applied, last_updated
    """
    delta = SUCCESS_DELTA if outcome == "success" else FAILURE_DELTA
    new_conf = max(MIN_CONFIDENCE, min(MAX_CONFIDENCE, pattern["confidence"] + delta))

    # consecutive_failures resets to 0 on success; increments on failure
    if outcome == "success":
        new_consec = 0
    else:
        new_consec = pattern["consecutive_failures"] + 1

    # Step 1: ENG-04 — demotion lock
    if pattern["tier"] == "demoted":
        new_tier = "demoted"
    # Step 2: ENG-05 — Gold exit guardrail (consec must reach threshold to exit)
    elif pattern["tier"] == "gold" and new_consec < GOLD_EXIT_THRESHOLD:
        new_tier = "gold"
    # Step 3: ENG-03 — natural tier from confidence
    else:
        new_tier = classify_tier(new_conf)

    return {
        **pattern,
        "confidence": new_conf,
        "tier": new_tier,
        "consecutive_failures": new_consec,
        "times_applied": pattern["times_applied"] + 1,
        "last_updated": datetime.now(timezone.utc).isoformat(),
    }
```

Key constraints:
- INIT_CONFIDENCE = 0.67 exactly (float literal)
- SUCCESS_DELTA = 0.05, FAILURE_DELTA = -0.10 exactly
- MAX_CONFIDENCE = 0.95, GOLD_EXIT_THRESHOLD = 3
- Evaluation order in apply_outcome: demotion lock → Gold guardrail → classify_tier (NEVER reorder)
- Returns {**pattern, ...} — spreads original dict and overrides changed fields
- Use datetime.now(timezone.utc).isoformat() NOT datetime.utcnow()
- No imports from akc.* — engine is self-contained
