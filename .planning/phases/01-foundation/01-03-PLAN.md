```python
import asyncio
import json
import os
import tempfile
from pathlib import Path

from akc.patterns.models import ConfidenceEvent, Pattern


class JsonlStore:
    """JSONL persistence layer for AKC patterns.

    STORE-03: All writes protected by a single asyncio.Lock.
    The ENTIRE read-modify-write cycle is held under the lock.
    """

    def __init__(self, kb_dir: str) -> None:
        # Security: canonicalize path to prevent path traversal via AKC_KB_DIR
        self._dir = Path(kb_dir).resolve()
        self._dir.mkdir(parents=True, exist_ok=True)
        self._lock = asyncio.Lock()
        self._patterns_path = self._dir / "patterns.jsonl"
        self._history_path = self._dir / "confidence_history.jsonl"

    # -------------------------------------------------------------------------
    # Sync helpers — called via asyncio.to_thread, never directly from async code
    # -------------------------------------------------------------------------

    def _read_patterns_sync(self) -> dict[str, dict]:
        """STORE-01: Read patterns.jsonl with last-write-wins dedup by id."""
        if not self._patterns_path.exists():
            return {}
        result: dict[str, dict] = {}
        with open(self._patterns_path, encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if line:
                    record = json.loads(line)
                    result[record["id"]] = record  # last occurrence wins
        return result

    def _write_patterns_atomic_sync(self, records: dict[str, dict]) -> None:
        """STORE-03: Write to tmp file → os.replace (atomic on POSIX).

        NEVER call this on confidence_history.jsonl — that file is append-only.
        """
        with tempfile.NamedTemporaryFile(
            mode="w",
            dir=self._dir,
            delete=False,
            suffix=".tmp",
            encoding="utf-8",
        ) as tmp:
            for record in records.values():
                tmp.write(json.dumps(record) + "\n")
            tmp_path = tmp.name
        os.replace(tmp_path, str(self._patterns_path))

    def _append_history_sync(self, event: dict) -> None:
        """STORE-02: Append to confidence_history.jsonl.

        Pure append — NEVER use os.replace on this file.
        No deduplication. Every event is preserved as an audit trail entry.
        """
        with open(self._history_path, "a", encoding="utf-8") as f:
            f.write(json.dumps(event) + "\n")

    # -------------------------------------------------------------------------
    # Public async interface — STORE-04
    # -------------------------------------------------------------------------

    async def load_stats(self) -> dict:
        """STORE-04: Return total pattern count and by_tier breakdown.

        Used by /health endpoint and startup logging (FNDTN-01, FNDTN-02).
        """
        async with self._lock:
            patterns = await asyncio.to_thread(self._read_patterns_sync)
        by_tier: dict[str, int] = {
            "gold": 0,
            "production": 0,
            "experimental": 0,
            "demoted": 0,
        }
        for p in patterns.values():
            tier = p.get("tier", "experimental")
            by_tier[tier] = by_tier.get(tier, 0) + 1
        return {"total": len(patterns), "by_tier": by_tier}

    async def load_active(
        self,
        min_tier: str = "production",
        tags: list[str] | None = None,
    ) -> list[dict]:
        """STORE-04: Return non-demoted patterns filtered by min_tier and tags.

        Results sorted by confidence descending (highest first).
        Demoted patterns are always excluded regardless of min_tier.
        """
        tier_rank = {"gold": 3, "production": 2, "experimental": 1, "demoted": 0}
        min_rank = tier_rank.get(min_tier, 2)

        async with self._lock:
            patterns = await asyncio.to_thread(self._read_patterns_sync)

        results = []
        for p in patterns.values():
            if p.get("tier") == "demoted":
                continue
            if tier_rank.get(p.get("tier", "experimental"), 0) < min_rank:
                continue
            if tags:
                pattern_tags = set(p.get("tags", []))
                if not any(t.lower() in pattern_tags for t in tags):
                    continue
            results.append(p)
        return sorted(results, key=lambda x: x.get("confidence", 0), reverse=True)

    async def update_pattern(self, pattern_id: str, outcome: str) -> None:
        """STORE-03: Read-modify-write under asyncio.Lock.

        The ENTIRE cycle (read → engine → write → append history) is held
        under self._lock. Do NOT release the lock between read and write.

        Args:
            pattern_id: ID of pattern to update
            outcome: "success" or "failure"
        """
        from akc.patterns import engine  # late import avoids circular dependency

        async with self._lock:
            patterns = await asyncio.to_thread(self._read_patterns_sync)
            if pattern_id not in patterns:
                return
            old = dict(patterns[pattern_id])  # snapshot before mutation
            patterns[pattern_id] = engine.apply_outcome(old, outcome)
            await asyncio.to_thread(self._write_patterns_atomic_sync, patterns)
            history_event = {
                "pattern_id": pattern_id,
                "outcome": outcome,
                "old_confidence": old["confidence"],
                "new_confidence": patterns[pattern_id]["confidence"],
                "old_tier": old["tier"],
                "new_tier": patterns[pattern_id]["tier"],
                "timestamp": patterns[pattern_id]["last_updated"],
            }
            await asyncio.to_thread(self._append_history_sync, history_event)

    async def save_pattern(self, pattern: Pattern) -> None:
        """Write a new or updated Pattern object to patterns.jsonl.

        Used by Phase 2 /remember endpoint to persist distilled patterns.
        Performs atomic read-modify-write under lock.
        """
        async with self._lock:
            patterns = await asyncio.to_thread(self._read_patterns_sync)
            patterns[pattern.id] = json.loads(pattern.model_dump_json())
            await asyncio.to_thread(self._write_patterns_atomic_sync, patterns)
```

Key constraints:
- asyncio.Lock acquired in update_pattern wraps the ENTIRE read → engine → write → append sequence
- _append_history_sync MUST use open(mode='a') — NEVER use os.replace on the history file
- _write_patterns_atomic_sync uses tempfile.NamedTemporaryFile + os.replace — NEVER write directly to patterns_path
- Path(kb_dir).resolve() for path canonicalization (security)
- `from akc.patterns import engine` is a LATE IMPORT inside update_pattern to avoid circular deps
- No caching of patterns in memory — always read from file (Phase 1 requirement)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| AKC_KB_DIR env var → Path | Path value crosses from env (controlled by deployment) to filesystem at __init__ |
| patterns.jsonl → _read_patterns_sync | File content from disk (may be externally modified) crosses into in-process dict |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | JsonlStore.__init__ kb_dir path | mitigate | `Path(kb_dir).resolve()` canonicalizes the path — eliminates `../` traversal via AKC_KB_DIR. mkdir(parents=True, exist_ok=True) confirms path is writable at startup |
| T-03-02 | Tampering | _read_patterns_sync — malformed JSONL line | accept | json.loads() raises json.JSONDecodeError on malformed lines; exception propagates to caller under asyncio.Lock. Phase 2 will add per-line try/except for corruption resilience — acceptable for Phase 1 where only trusted writes produce the file |
| T-03-03 | Tampering | _write_patterns_atomic_sync — tmp file in same dir | accept | tempfile.NamedTemporaryFile with suffix=".tmp" in same directory as target ensures os.replace is atomic (same filesystem). Tmp file is cleaned up by os.replace on success or will be orphaned on crash (acceptable — small .tmp file) |
| T-03-04 | Denial of Service | asyncio.Lock contention | accept | Single service, low write volume in hackathon scope. Lock held during file I/O via asyncio.to_thread — does not block the event loop itself |
</threat_model>
